221:         $query = '';
222:         foreach ($values as $field => $value) {
223:             if (is_array($value)) {
224:                 $value = array_map(array($this, 'escape'), $value);
225:                 $query.= $field.' IN('.implode(',', $value).')'.' '.$separator.' ';
